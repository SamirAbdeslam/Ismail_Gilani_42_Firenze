#include "libft.h"

void	ft_putchar(char c)
{
	write (1, &c, 1);
}

void	ft_putnbr_fd(int n, int fd)
{

	if (n == -2147483648)
	{
		ft_putchar('-');
		ft_putchar('2');
		ft_putnbr(147483648);
	}
	else if (n < 0)
	{
		ft_putchar('-');
		n = -n;
		ft_putnbr(n);
	}
	else if (n >= 0 && n <= 9)
	{
		n = n + '0';
		ft_putchar(n);
	}
	else
	{
		ft_putnbr(n / 10);
		ft_putnbr(n % 10);
	}

	write(1, n, )
}
